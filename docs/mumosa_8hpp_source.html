53template<typename Index_, typename Float_>
54std::pair<Float_, Float_> compute_distance(Index_ num_cells, Float_* distances) {
55 Float_ med = tatami_stats::medians::direct(distances, num_cells, /* skip_nan = */ false);
56 Float_ rmsd = 0;
57 for (Index_ i = 0; i < num_cells; ++i) {
58 auto d = distances[i];
59 rmsd += d * d;
60 }
61 rmsd = std::sqrt(rmsd);
62 return std::make_pair(med, rmsd);
63}

78std::pair<Float_, Float_> compute_distance(const knncolle::Prebuilt<Dim_, Index_, Float_>& prebuilt, const Options& options) {
79 size_t nobs = prebuilt.num_observations();
80 std::vector<double> dist(nobs);
81
82 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
83 auto searcher = prebuilt.initialize();
84 std::vector<Float_> distances;
85 for (size_t i = start, end = start + length; i < end; ++i) {
86 searcher->search(i, options.num_neighbors, NULL, &distances);

91 }, nobs, options.num_threads);

142template<typename Float_>
143Float_ compute_scale(const std::pair<Float_, Float_>& ref, const std::pair<Float_, Float_>& target) {
144 if (target.first == 0 || ref.first == 0) {
145 if (target.second == 0) {
146 return std::numeric_limits<Float_>::infinity();
147 } else if (ref.second == 0) {
148 return 0;
149 } else {
150 return ref.second / target.second;
151 }
152 } else {
153 return ref.first / target.first;
154 }

170template<typename Float_>
171std::vector<Float_> compute_scale(const std::vector<std::pair<Float_, Float_> >& distances) {
172 std::vector<Float_> output(distances.size());
173
174 // Use the first entry with a non-zero RMSD as the reference.
175 bool found_ref = false;
176 size_t ref = 0;
177 for (size_t e = 0; e < distances.size(); ++e) {
178 if (distances[e].second) {
179 found_ref = true;
180 ref = e;
181 break;
182 }
183 }
184
185 // If all of them have a zero RMSD, then all scalings are zero, because it doesn't matter.
186 if (found_ref) {
187 const auto& dref = distances[ref];
188 for (size_t e = 0; e < distances.size(); ++e) {
189 output[e] = (e == ref ? 1 : compute_scale(dref, distances[e]));
190 }
191 }
192
193 return output;
194}
